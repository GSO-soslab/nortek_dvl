cmake_minimum_required(VERSION 2.8.3)
project(nortek_dvl)

add_compile_options(-std=c++11)

find_package(catkin REQUIRED COMPONENTS
  roscpp
  rospy
  std_msgs
  serial
  geometry_msgs
  message_generation
)

add_message_files(
  FILES
  DVL.msg
  CurrentProfile.msg
  CellMeasure.msg
)

generate_messages(
  DEPENDENCIES
  std_msgs
  geometry_msgs
  nortek_dvl # for new defined msg
)


catkin_package(
 INCLUDE_DIRS include
 #LIBRARIES nortek_dvl
 CATKIN_DEPENDS roscpp rospy std_msgs serial geometry_msgs message_runtime
 DEPENDS system_lib
)

###########
## Build ##
###########


include_directories(
  include
  ${catkin_INCLUDE_DIRS}
)

# build library
add_library(dvl_driver
  src/dvl_driver.cpp
)

target_link_libraries(dvl_driver
  ${catkin_LIBRARIES}
)

# build driver
add_executable(driver_node src/driver_node.cpp)

target_link_libraries(driver_node
  dvl_driver
  ${catkin_LIBRARIES}
)
add_dependencies(driver_node ${PROJECT_NAME}_generate_messages_cpp)

## Test code

  ${catkin_LIBRARIES}
)

  ${catkin_LIBRARIES}
